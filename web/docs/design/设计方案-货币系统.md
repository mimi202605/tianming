## Ⅲ 铸币全模型详方

### MintAgency 结构

```js
{
  id, name: '宝泉局', location: region_id,
  type: 'central' | 'provincial' | 'private_licensed' | 'private_illegal',
  capacity: 100000,               // 每回合最大铸币能力
  staffing: 80,                    // 工匠充足度 0-100
  coinType: 'copper',              // 铸什么币
  purityStandard: 1.0,             // 本机构成色标准
  costPerUnit: 0.7,                // 铸 1 单位钱需要原料成本（文 → 需 0.7 文铜料）
  seignioragePerUnit: 0.3,         // 铸币利润
  enabled: true
}
```

### 每回合铸币流程

```js
function _mintCycle() {
  for (let agency of GM.currency.mintAgencies) {
    if (!agency.enabled) continue;
    var rawNeeded = agency.capacity * agency.costPerUnit;
    var rawAvailable = GM.currency.coins[agency.coinType].rawReserve;
    if (rawAvailable < rawNeeded || agency.staffing < 30) continue;
    var actualOutput = agency.capacity * (agency.staffing/100) * (1 - agency.purityStandard === 1 ? 0 : 0.1);
    GM.currency.coins[agency.coinType].stock += actualOutput;
    GM.currency.coins[agency.coinType].rawReserve -= rawNeeded;
    GM.currency.coins[agency.coinType].mintQuantity += actualOutput;
    Fiscal.add('guoku.money', actualOutput * agency.seignioragePerUnit, '铸币利润');
    agency.staffing = Math.max(0, agency.staffing - 1);   // 每回合耗损
  }
}
```

### 成色降级 debasementLevel

- 诏令"减重改铸" → debasementLevel += 0.1
- 效果：铸币产量+20%，但 purchasingPowerFactor × 0.9
- 连续降级 → 民众抵制 → 劣币驱逐良币（格雷欣法则）
- debasementLevel > 0.5 → 触发"恶钱不行"事件

### 私铸 privateMintingLevel

```js
// 每回合计算
privateMintShare = clamp(
  0.05                                                     // 基础
  + currency.debasementLevel * 0.3                         // 官钱越劣，私铸越诱人
  + currency.metalShortage * 0.2                           // 金属料短缺催生私铸
  + (1 - currency.mintingControl) * 0.3                    // 查禁不严
  + region.poverty * 0.2,                                  // 穷乡僻壤易私铸
  0, 0.9
);
```

- 私铸币以劣币形式进入流通（purity × 0.5-0.7）
- 玩家诏令"严禁私铸" → `mintingControl += 0.1`，持续消耗财政
- `mintingControl` 每回合自然衰减 0.02（查禁需不断投入）

### 铸币利润 seigniorage

```js
totalSeigniorage = Σ(agency.output × (faceValue - costPerUnit))
Fiscal.add('guoku.money', totalSeigniorage, '铸币利润')
```

降低 purity → 短期 seigniorage 上升，长期 purchasingPowerFactor 崩坏。

### 钱荒 / 钱贱

```js
economicActivity = GM.vars.pop * baseActivityPerCapita + tradeVolume
moneySupply = currency.copper.stock * (1 - currency.copper.hoardingPressure)
moneySupplyRatio = moneySupply / economicActivity

if (moneySupplyRatio < 0.6) {
  // 钱荒（钱贵物贱）
  event: "钱贵物贱，商贸滞塞"
  market.grainPrice *= 0.85
  unrest += 2
}
if (moneySupplyRatio > 1.5) {
  // 钱贱（通胀）
  event: "铜钱泛滥"
  market.grainPrice *= 1.2
}
```

### 铜钱外流

宋代铜钱大量流入日本/高丽/南洋，每回合：
```js
currency.copper.stock -= currency.copper.outflowRate * currency.copper.stock
// outflowRate 默认 0.02，海贸发达时 0.05
```

玩家可"禁钱出海" → outflowRate 降 50%，但海贸萎缩。

### 六状态生命周期

| 状态 | 触发条件 | 信用 | 通胀贡献 | 特征 |
|------|---------|------|---------|------|
| issue | 首发 | 100 | 0 | 新鲜、稀缺、抢手 |
| circulate | 稳定 | 80-100 | 0-5% | 准备金充足，商贸接受 |
| overissue | reserveRatio < 0.2 | 60-80 | 5-20% | 发行过量初显 |
| depreciate | reserveRatio < 0.1 | 30-60 | 20-100% | 明显贬值 |
| collapse | reserveRatio < 0.05 或 credit < 20 | <30 | >100% | 信用崩溃，百姓拒用 |
| abolish | 朝廷废止 | 0 | — | 退出流通，按比例兑新币或作废 |

### 状态转移规则

```js
function _updatePaperState(issuance) {
  // 计算 reserveRatio
  issuance.reserveRatio = _calcReserveValue(issuance.reserveBacking) 
                        / issuance.currentCirculation;
  
  // 更新 creditLevel
  var creditDrift = 0;
  if (issuance.reserveRatio < 0.1) creditDrift = -5;
  else if (issuance.reserveRatio < 0.2) creditDrift = -2;
  else if (issuance.reserveRatio > 0.4) creditDrift = +1;
  creditDrift -= issuance.cumulativeIssued / issuance.originalAmount * 0.5;  // 累计滥发扣分
  issuance.creditLevel = clamp(issuance.creditLevel + creditDrift, 0, 100);
  
  // 状态转移
  if (issuance.reserveRatio < 0.05 || issuance.creditLevel < 20) {
    _transitionState(issuance, 'collapse');
  } else if (issuance.reserveRatio < 0.1) {
    _transitionState(issuance, 'depreciate');
  } else if (issuance.reserveRatio < 0.2) {
    _transitionState(issuance, 'overissue');
  }
  
  // 计算通胀贡献
  issuance.inflationCarried = Math.max(0, 1 - issuance.reserveRatio) * 
                              (issuance.currentCirculation / issuance.originalAmount);
}
```

### 粮价计算（每回合）

```js
function _updateGrainPrice() {
  var supply = Σ(region.farmland * yield * (1-disaster) * (1-logisticsLoss));
  var demand = totalPop * capitaConsumption + militaryGrain + stockpiling;
  var ratio = demand / Math.max(1, supply);
  
  var basePrice = 100;  // 文/石
  market.grainPrice = basePrice 
    * Math.pow(ratio, priceElasticity)   // 弹性系数 1.5
    * market.warInflation
    * market.seasonalFactor
    * market.yearFortune
    * (1 + market.speculationLevel * 0.3);
}
```

同理计算 clothPrice、saltPrice、各金属料价。

### 地域价差与商贸套利

```js
// 每两区之间价差
var priceGap = region2.grainPrice - region1.grainPrice;
var transportCost = baseCostPerLi * distance * (1 + banditry) * (1 + disruption);
var arbitrageProfit = priceGap - transportCost;

if (arbitrageProfit > 0) {
  // 商人 NPC 自动套利
  var flowAmount = Math.min(region1.grainSurplus, arbitrageProfit * arbitrageIntensity);
  region1.grainStock -= flowAmount;
  region2.grainStock += flowAmount;
  // 套利缩小价差
  region1.grainPrice += flowAmount * priceImpact;
  region2.grainPrice -= flowAmount * priceImpact;
}
```

### 运输成本

```js
function transportCost(from, to, goodType) {
  var distance = getDistance(from, to);
  var base = baseCostPerLi * distance;
  var mod = (1 + region.banditry) * (1 + region.disruption);
  var typeFactor = {
    'grain_by_canal': 0.3,    // 漕运最便宜
    'grain_by_land': 1.0,
    'grain_by_sea': 0.2,      // 海运最便宜但风险高
    'silver_by_land': 0.5,     // 银重量小
    'paper_by_land': 0.05      // 纸币轻如鸿毛
  }[goodType];
  return base * mod * typeFactor;
}
```

### 改革执行流程

1. 玩家写"推行 X 改革" → AI 识别类型
2. 检查 requirements（朝代阶段/宰相能力/财政基础等）
3. 自动生成廷议议题 → 各派立场
4. 廷议通过 → 进入 transitionYears 年过渡期
5. 过渡期每回合触发阻力/配合事件
6. 过渡结束计算最终成败：
   ```
   finalRate = baseSuccessRate 
             × (0.7 + 0.3 × 宰相能力/100)
             × (0.8 + 0.2 × 财政稳健度)
             × (1 - 民间接受度阻力)
   ```
7. 成 → 应用 effects；败 → 回滚，代价可能极大（王莽式）

## Ⅷ 地域币值差异详方（接受度系数 0-1）

### region.currencyProfile 结构

```js
{
  preferredCoin: 'copper',                   // 主流偏好
  acceptance: {                               // 各币种接受度 0-1
    gold: 0.3,
    silver: 0.7,
    copper: 1.0,
    iron: 0.0,                                // 不接受（非铁钱区）
    shell: 0.0,
    paper_jiaozi: 0.9,                        // 按 issuance id 细分
    paper_huizi: 0.6
  },
  exchangeRateVsStandard: {                   // 各币种对京师标准的本地汇率
    silver: 1.0, copper: 1.0, iron: 0.1, paper: 0.95
  },
  localPurity: {                              // 本地币种成色
    silver: 0.93, copper: 0.95
  },
  privateMintingLevel: 0.1,
  localMintAgency: null,                      // 本地有无铸币机构
  regionalReform: null                         // 本地币制特例（如藩镇独铸）
}
```

### 接受度的作用

```js
// 实际流通量 = 理论流通 × 本地接受度
function localEffectiveStock(region, coinType) {
  return GM.currency.coins[coinType].stock * region.currencyProfile.acceptance[coinType];
}

// 跨区贸易需按接受度折算
function payInRegion(region, coinType, amount) {
  if (region.currencyProfile.acceptance[coinType] < 0.3) {
    return { accepted: false, reason: '本地不接受' + coinType };
  }
  var effectiveValue = amount 
    * region.currencyProfile.acceptance[coinType]
    * region.currencyProfile.exchangeRateVsStandard[coinType];
  return { accepted: true, effectiveValue };
}
```

### 经典地域设定（剧本默认）

| 地区 | preferredCoin | 备注 |
|------|--------------|------|
| 京畿 | 按朝代变 | acceptance 各币种都接近 1.0，纯度最高 |
| 江南 | silver（明清）/ copper（宋元前）| 商业发达，纸币接受度高 |
| **四川（宋）**| **iron** | **acceptance.copper = 0.3, iron = 1.0（独铁钱区）** |
| 两淮（南宋）| iron | 南宋末设铁钱 |
| 岭南 | silver | 海贸银流入，silver purity 偏低 0.85 |
| 河北 | copper | 军费消耗地，纸币接受度低 |
| 河西/陇右 | 实物 | acceptance 低，多为粮布实物 |
| 安南/交趾 | 铜/布 | 局部流通 |
| 云南 | 贝+布（明早）| 实物交易为主 |
| 辽东 | 随朝代变 | 辽金元清独特 |
| 藩镇（唐后期）| 自铸 | acceptance.copper 按藩镇本地 |

### 跨地域商贸与货币

商人套利会同时考虑**价差 + 币种接受度 + 汇率**：

```js
// 江南粮食运陕西，用银付款
netProfit = priceGap - transportCost - 币种兑换损耗
// 币种兑换损耗 = amount × (1 - 陕西.acceptance.silver)
```

### 统一钱法（强制收敛）

玩家诏令"统一钱法" → 逐年降低各地方差：
```js
region.acceptance[coinType] → 向全国均值收敛（每回合 +0.02）
```

高阻力（藩镇强烈反对）、高成本（需派督察官）、耗时（10+ 回合）。

- 墨西哥银→马尼拉→福建月港
- 明后期年均 50-150 万两（部分 300 万两）

- 1720s 年均 50 万两
- 1750s 100-150 万两

### 9.1 通胀 → 民心（粮价传导）

通货膨胀通过粮价/物价传导到民心（见民心 §2.6 priceStability）：

```javascript
function applyCurrencyInflationToMinxin() {
  var purchasingPower = GM.currency.actualPurchasingPower;
  var baseline = GM.currency.baselineValue;
  var inflation = baseline / purchasingPower;  // 通胀率
  
  if (inflation <= 1.2) return +1;  // 轻微稳定
  return -Math.pow(inflation - 1.2, 1.8) * 10;
  //   1.5 倍 → -5；2 倍 → -13；3 倍 → -32
}
```

**历史参照**：金代交钞末期通胀超 1000 倍 → 民心崩盘 → 亡金。

### 9.2 滥发纸币 → 皇威（货币信用崩溃）

纸币滥发是历代皇威受损的重要来源：

```javascript
function onPaperCurrencyAbuse(ratio) {
  // ratio：实际发行 / 准备金倍数
  if (ratio > 3) {
    GM.huangwei.index -= 5;
    GM.huangwei.drains.diplomaticHumiliation += 5;  // 外邦知中土滥发
  }
  if (ratio > 10) {
    GM.huangwei.index -= 15;
    GM.huangwei.subDims.foreign.value -= 20;  // 外邦不再接受朝贡贸易
  }
  if (ratio > 50) {
    GM.huangwei.index -= 30;
    triggerEvent('currencyCollapse');
  }
}
```

**历史参照**：元末滥发至元宝钞、钞法完全失信、明初罢钞。

### 9.3 币制改革 × 皇权需求

重大币制改革需要皇权支撑：

```javascript
function evaluateCurrencyReformFeasibility(reform) {
  var h = GM.huangquan.index;
  var w = GM.huangwei.index;
  
  var reqByScale = {
    'minor':       { h: 30, w: 30 },   // 如铸新钱
    'moderate':    { h: 50, w: 50 },   // 如发行新钞
    'major':       { h: 65, w: 60 },   // 如一条鞭法
    'revolutionary': { h: 80, w: 70 }  // 如废除旧币
  };
  
  var req = reqByScale[reform.scale];
  if (h < req.h) return { feasible: false, reason: '皇权不足以推行' };
  if (w < req.w) return { feasible: false, reason: '皇威不足，地方/民间不信' };
  
  return { feasible: true };
}
```

### 9.4 私铸 × 腐败（监管失效）

私铸泛滥与腐败强相关：

```javascript
function calcPrivateMintingRate() {
  var baseRate = 0.05;  // 5% 基础私铸率
  var localCorr = GM.corruption.subDepts.provincial.true;
  var fiscalCorr = GM.corruption.subDepts.fiscal.true;
  
  // 腐败高 → 监管失效 → 私铸猖獗
  var corrEffect = (localCorr + fiscalCorr) / 200 * 0.3;
  
  // 皇权弱 → 地方默许
  if (GM.huangquan.index < 40) corrEffect += 0.05;
  
  return baseRate + corrEffect;
}
```

**历史参照**：明末民间私铸"沙板钱"几乎占市场半数。

### 9.7 白银流入皇威联动

大量白银流入（如明代美洲银元）带来皇威加成：
- 国际贸易繁荣 → 皇威加成（外邦敬畏）
- 但过度依赖外银 → 白银外流时皇威反噬

```javascript
function applySilverFlowToHuangwei() {
  var netInflow = GM.silverInflow - GM.silverOutflow;
  if (netInflow > 10000000) {  // 大量流入
    GM.huangwei.subDims.foreign.value += 3;
  } else if (netInflow < -10000000) {  // 大量流出（鸦片期）
    GM.huangwei.subDims.foreign.value -= 10;
    GM.huangwei.drains.diplomaticHumiliation += 10;
  }
}
```
