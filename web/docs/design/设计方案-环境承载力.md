- 挂在户口系统 hover 展开，不占顶栏

## Ⅰ 五维承载力详方

### ① 土地支撑 farmlandSupport

```js
function _calcFarmlandSupport(region) {
  return region.arableArea
    × region.soilFertility                   // 地力（会衰退）
    × _techYieldMult(region)                  // 技术加成
    × _cropMix(region)                        // 作物组合
    × _climateFactor(region)                  // 气候
    × _irrigationBonus(region)                // 灌溉加成
    - region.ecoScars.salinization × factor   // 盐碱化损失
    - region.ecoScars.soilErosion × factor;   // 水土流失损失
}
```

| 明 | 1.8 | 2.2 | — |

### ② 水资源 waterSupport

```js
function _calcWaterSupport(region) {
  return (region.surfaceWater + region.aquifer + region.irrigation)
    × region.waterDistributionEfficiency
    × _monsoonFactor(region, year)
    - region.ecoScars.waterTableDrop × factor
    - region.ecoScars.riverSilting × factor;
}
```

**区域水资源类型**：

| 类型 | 稳定度 | 可耗竭 | 典型 |
|------|-------|-------|------|
| 常年河 | 0.9 | 否 | 长江、珠江、辽河 |
| 季节河 | 0.5 | — | 黄河中下游、汉水 |
| 绿洲 | 0.3 | 易 | 河西走廊、西域 |
| 地下水 | 0.7 | 易 | 华北平原、关中 |
| 湖泊 | 0.8 | 可变 | 洞庭、鄱阳、太湖 |

### ③ 燃料支撑 fuelSupport

```js
function _calcFuelSupport(region) {
  var forestFuel = region.forestArea × woodYieldPerUnit;
  var coalFuel = region.coalReserve × region.techLevel.mining × _mineRate;
  var strawFuel = region.farmlandArea × strawYieldPerMu;
  
  var demand = region.mouths × capitaFuelNeed    // 约 2 石/年
             + region.cities × urbanFuelBonus      // 城市额外
             + region.industry × industrialFuelNeed; // 冶铁制瓷
             
  return forestFuel + coalFuel + strawFuel - demand;
}
```

**历史燃料危机节点**：
- 1000 年前后：汴京木柴紧张，煤炭兴起
- 1700 年前后：华北森林几尽，全面煤炭化
- 1850 年前后：江南燃料占家庭开支 15%

```js
function _calcHousingSupport(region) {
```

### ⑤ 卫生支撑 sanitationSupport

```js
function _calcSanitationSupport(region) {
  var baseSupport = region.cleanWaterAccess
                  × region.sewageDisposal
                  × region.fecalManagement;  // 粪肥返田是古代特有优势
                  
  var urbanDensityPenalty = region.urbanDensity > 10000 ? 0.3 : 0;
  
  return baseSupport - urbanDensityPenalty - region.ecoScars.urbanSewageOverload;
}

// 城市密度触发事件
if (urbanDensity > 10000人/方里) {
  smallpoxRiskMult += 0.3;
  choleraRiskMult += 0.5;
  plagueRiskMult += 0.4;
}
```

### 承载力取小值

```js
region.carryingMax = Math.min(
  farmlandSupport,
  waterSupport,
  fuelSupport,
  housingSupport,
  sanitationSupport
```

## Ⅱ 生态疤痕累积机制

```js
foreach region:
  foreach scar in ecoScars:
    var usage = _getUsageRate(region, scar.type);    // 当前消耗速率
    var sustainable = _getSustainableRate(region, scar.type);
    
    if (usage > sustainable) {
      scar.value += (usage - sustainable) × accumRate × (1 + severity);
    } else if (region.policies.conservation[scar.type]) {
      scar.value -= recovery × policyStrength × (1 - scar.value);  // 越深越难恢复
    }
    
    // 疤痕影响承载力
    region.carrying.affectedByScars = baseCapacity × (1 - scar.value × impact);
```

**疤痕历史节点示例**：
- 周秦黄河流域森林 50%+ → 清末 10%（deforestation 从 0 → 0.8）
- 关中唐中期后盐碱化加剧（salinization 至 0.3-0.5）
- 鄂尔多斯明末屯田后 → 清中沙漠化（desertification 0 → 0.6）
- 华北明清地下水位下降（waterTableDrop 0 → 0.4）

## Ⅲ 过载多级反馈（关键机制）

```js
function _applyOverloadEffects(region) {
  var load = region.currentLoad;
  
  if (load < 0.8) return;   // 安全区
  
  // 各级触发
  if (load > 1.0) {
    region.populationDynamics.deathRateMult *= 1.0 + (load - 1.0) × 0.5;
    region.populationDynamics.birthRateMult *= Math.max(0.5, 2 - load);
    region.migration.outRate += (load - 1.0) × 0.03;
  }
  
  if (load > 1.2) {
    region.plaguRisk += 0.1;          // 疫病易发
    region.banditryRisk += 0.1;       // 盗贼兴起
  }
  
  if (load > 1.3) {
    region.famineEvent = { probability: (load - 1.3) × 2 };
    region.peasantResentment += 5;
  }
  
  if (load > 1.4) {
    region.rebellionRisk += (load - 1.4) × 0.3;
    _triggerEvent('pop_overload_crisis', region);
  }
  
  // 连续过载更严重
  if (region.overloadYears > 5) {
    region.ecoDebt += overloadYears × 0.1;
    _triggerEvent('dynasty_cycle_peak_overload', region);
  }
}
```

**王朝周期律的生态表现**：
- 建朝初（load 0.3-0.5）：人少地多，新开垦，生态恢复
- 朝代中期（0.6-0.9）：人口增长，开垦扩张，生态稳定
- 朝代末期（>1.0）：过载、饥荒、逃户、起义
- 战乱后人口骤降（回到 0.3）：生态部分恢复（forest regrowth）

## 五维联动公式（核心）

```js
// 每年更新
function _annualEnvUpdate(region) {
  // 1. 消耗
  region.forestArea -= region.fuelDemand / woodYieldPerUnit;
  region.aquiferLevel -= region.waterDemand / region.rechargeRate;
  region.soilFertility -= (1 - region.fertilization) × decay;
  
  // 2. 疤痕累积
  foreach scar: _updateScar(scar);
  
  // 3. 重新计算承载力
  region.carrying = _recalcAllDimensions(region);
  region.carryingMax = Math.min(...region.carrying.values());
  
  // 4. 加载比
  region.currentLoad = region.mouths / region.carryingMax;
  
  // 5. 过载反馈
  _applyOverloadEffects(region);
  
  // 6. 危机事件概率检测
  _checkEnvironmentalCrises(region);
  
  // 7. 政策效果
  _applyConservationPolicies(region);
  
  // 8. 技术进步解锁
  _checkTechUnlock(region, currentYear);
}
```

## 与其他子系统联动

| 对方系统 | 联动点 |
|---------|-------|
| 在籍户口 | 过载触发逃户/死亡率+/生育率-；承载力是人口动态核心制约 |
| 徭役 | 大工程（修堤/疏浚）直接影响疤痕；徭役过重 + 过载 → 民变爆发 |
| 兵役 | region 贫瘠则兵员质量差；军屯既可开荒也可加剧水土流失 |

| 经济·货币 | 铜矿枯竭 → 钱荒；铁矿枯竭 → 工具贵 |
| 气候系统（F1.1）| 气候直接影响各维承载力 |
| 作物革命（F1.2）| 新作物提高 farmlandSupport |
| 疫病（F1.3）| 卫生过载 → 疫病频率高 |
| AI 推演 | prompt 注入生态状态（古语）|

### 9.1 生态疤痕 → 民心（直接传导）

生态恶化直接影响民心（民心 §2.7 security + §2.10 heavenSign）：

```javascript
function applyEcologicalScarsToMinxin() {
  for (var scar of GM.ecologicalScars) {
    var region = scar.region;
    var severity = scar.severity;
    
    // 水土流失、沙化、水源枯竭 → 民心渐降
    if (scar.type === 'weakDike' || scar.type === 'soilErosion') {
      GM.minxin.byRegion[region].true -= severity * 0.5;
    }
    
    // 生态大灾（蝗灾、瘟疫）→ 天象级效果
    if (scar.type === 'locust' || scar.type === 'plague') {
      GM.minxin.omens.heavenlySigns.push({
        type: 'ecologicalDisaster',
        severity: severity,
        turn: GM.turn
      });
    }
  }
}
```

```javascript
function advanceTechLadder(tech) {

}
```
