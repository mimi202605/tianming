## 资源一：公库（机构绑定 · 只读镜像）

### 设计哲学修订
**公库不属于个人，而属于机构**（行政区划/中央部门/军事单位/内廷机构）。角色面板显示的公库仅是**对所绑定机构的只读镜像**。任职时激活镜像、离任时归零、前任亏空由后任继承。

### 数据模型

```js
// char 层：仅存"绑定引用"，非存自有数据
char.resources.publicTreasury = {
  binding: char.currentPosition?.publicTreasuryBinding || null
  // 不存 quota/used/available 等字段！这些属于机构层
}

// 显示函数（从机构实时读取）
function getCharPublicTreasuryDisplay(char) {
  if (!char.resources.publicTreasury.binding) {
    return { money: 0, grain: 0, cloth: 0 };  // 闲职或无官 → 全零
  }
  var entity = _resolveBinding(char.resources.publicTreasury.binding);
  return {
    money: entity.publicTreasury.money,
    grain: entity.publicTreasury.grain,
    cloth: entity.publicTreasury.cloth
  };
}
```

### 机构层公库字段

公库真正的所有者是机构（region/ministry/military/imperial），每个机构有独立的 `publicTreasury`：

```js

  previousHead: char.name,    // 前任
  handoverLog: [              // 交接历史
    { turn, fromChar, toChar, surplus, deficit, notes }

// 中央部门：guoku 下子预算
GM.fiscal.guoku.subBudgets = {
```

### 任命/离任钩子

```js
function onAppointment(char, position) {
  var oldBinding = char.resources.publicTreasury.binding;
  var newBinding = position.publicTreasuryBinding;
  
  // 解除旧绑定
  if (oldBinding) {
    var oldEntity = _resolveBinding(oldBinding);
    oldEntity.handoverLog.push({
      turn: GM.turn,
      fromChar: char.name,
      toChar: null,
      surplus: oldEntity.publicTreasury.money.available > 0 ? {...} : {},
      deficit: oldEntity.publicTreasury.money.deficit,
      notes: '转任'
    });
    oldEntity.currentHead = null;
  }
  
  // 建立新绑定
  char.resources.publicTreasury.binding = newBinding;
  char.currentPosition = position;
  
  if (newBinding) {
    var newEntity = _resolveBinding(newBinding);
    newEntity.currentHead = char.name;
    newEntity.headSinceTurn = GM.turn;
    
    // 前任亏空继承
    if (newEntity.publicTreasury.money.deficit > 0) {
      _triggerEvent('inherit_deficit', { 
        newHead: char.name, 
        amount: newEntity.publicTreasury.money.deficit,
        fromPrevious: newEntity.previousHead
      });
    }
  }
}

function onDismissal(char) {
  var binding = char.resources.publicTreasury.binding;
  if (binding) {
    var entity = _resolveBinding(binding);
    entity.handoverLog.push({
      turn: GM.turn,
      fromChar: char.name,
      toChar: null,
      surplus: entity.publicTreasury.money.available > 0 ? {...} : {},
      deficit: entity.publicTreasury.money.deficit,
      notes: '免职/致仕'
    });
    entity.previousHead = char.name;
    entity.currentHead = null;
  }
  char.resources.publicTreasury.binding = null;
  char.currentPosition = null;
  // char 面板公库自动归零（绑定解除）
}
```

### 动态机构（玩家运行时新设）

```js
// 玩家通过"官制设立诏书"新设的部门/军单位/行政区划
GM.dynamicInstitutions = {
  ministries: { 'haifang': {...} },
  positions:  { '海防尚书': { publicTreasuryBinding: 'ministry:haifang', ... } },
  regions:    {},
  militaryUnits: {}
}

// 新增职位也走统一绑定解析
function _resolveBinding(binding) {
  var [type, id] = binding.split(':');
  switch(type) {
    case 'region':   return GM.regionMap[id] || GM.dynamicInstitutions.regions[id];
    case 'ministry': return GM.fiscal.guoku.subBudgets[id] 
                         || GM.dynamicInstitutions.ministries[id];
    case 'military': return GM.fiscal.guoku.subBudgets.military[id]
                         || GM.dynamicInstitutions.militaryUnits[id];
    case 'imperial': return GM.fiscal.neicang.subBudgets[id];
    default: return null;
  }
}
```

### 配额生成
任命时 `_assignPublicTreasury(char, position)`：
```js
quota = position.baseTreasury                        // 职位基准
      × (1 + 朝代系数)
      × (1 + 地域重要度 fossil for 地方官)
      × (1 + 特殊授权 for 钦差/总督)
```

### 去职/调任
去职 → 结转：
- available 正 → 盈余移交新任（或退还中央）
- available 负（亏空） → 触发"追亏"事件，扣新任公库或本人私产补

### 挪用贪污
支出时玩家/AI 可私挪 → 记入 `illicit` 分类：
- 私挪金额计入 `char.privateWealth.cash`
- 公库 `used` 增加（占用 quota）
- 隐蔽度取决于 `concealmentSkill`
- 御史查账发现差额 → `auditResult='flagged'` → 腐败调查


### 汇总估值

```js
function _calcPrivateSummary(char) {
  var total = { money:0, grain:0, cloth:0 };
  // 现金三列直接加
  total += char.privateWealth.cash;
  // 田产按市价折
  char.privateWealth.land.forEach(l => {
    total.money += l.area * GM.currency.market.landPricePerUnit;
    total.grain += l.yieldPerYear.grain;  // 年产粮
  });
  // 房产、商铺、珍玩、家族企业均折为 money
  ...
  // 扣除债务
  char.privateWealth.debts.forEach(d => {
    total.money -= d.amount.money;
    total.grain -= d.amount.grain;
    total.cloth -= d.amount.cloth;
  });
  return total;
}
```

### 负值场景
- 俸薄借债：明代贫官 privateWealth.summary.money 负值
- 破产商人：债务 > 资产
- 失势贵族：抄家后剩残局

```js
function calcPromotionChance(char, targetPosition) {
  var basedOnTier = {
    '初入仕途': 0.05,
    '颇有政绩': 0.15,
    '地方重臣': 0.35,
    '朝廷要员': 0.55,
    '国之栋梁': 0.75,
    '千古名臣': 0.90
```

| 朝代 | 默认 mode | 理由 |
|------|----------|------|
| 先秦 | communal | 宗法制 |
| 汉 | communal | 同居共财为主 |

| 宋 | divided | 分家普遍 |

### 共财模式运算
- 个人 `privateWealth` 计入 `family.sharedWealth` 上层账
- 日常支出从共财出，个人赏赐入共财
- 继承按 `clanRules.inheritance` 分

### 分家模式
- 每人独立 `privateWealth`
- 仅祠堂、族田归家族共有
- 继承按嫡庶长幼

## 抄家清算（含隐匿挖掘 + 亲族株连）

### 抄家流程

```js
function confiscate(char, policy) {
  var declared = sumDeclaredWealth(char);
  var hidden = estimateHiddenWealth(char);
  
  var excavationRate = policy.thoroughness * 0.3
                     + policy.interrogationPressure * 0.5
                     + policy.informantQuality * 0.2;
  var excavated = hidden * clamp(excavationRate, 0, 1);
  
  // 亲族株连
  var clanPenalty = 0;
  if (policy.clanImplication === 'nine_generations') {
    familyMembers.forEach(m => {
      confiscate(m, { ...policy, clanImplication:'none' });  // 递归不再株连
    });
  } else if (policy.clanImplication === 'immediate_family') {
    directRelatives.forEach(m => confiscate(m, policy));
  }
  
  // 收益归属
  var targetAccount = policy.destination || 'neicang';
  Fiscal.add(targetAccount + '.money',  declared.money + excavated.money,  '抄' + char.name + '家');
  Fiscal.add(targetAccount + '.grain',  declared.grain + excavated.grain,  '抄家');
  Fiscal.add(targetAccount + '.cloth',  declared.cloth + excavated.cloth,  '抄家');
  
  // 抄后角色状态
  char.resources = _emptyResources();
  char.socialClass = 'commoner';
```

### 隐匿估值

```js
function estimateHiddenWealth(char) {
  var visibleTotal = sumDeclaredWealth(char);
  var concealmentRatio = 
      char.clanInfluence * 0.3       // 豪门易隐
    + (1 - char.monitoring) * 0.3     // 监察弱
    + char.corruption * 0.4           // 越贪越擅藏
    + (char.resources.virtueMerit > 500 ? 0.2 : 0);  // 权贵藏匿能力强
  return visibleTotal * concealmentRatio;
}
```

```js
function _estimateResources(char, sc) {
  var dynastyBase = DYNASTY_FISCAL_BASE[sc.dynasty] || DEFAULT_BASE;
  var rankMultiplier = RANK_COEF[char.rank];            // 一品 1.0 → 九品 0.1
  var originMultiplier = ORIGIN_COEF[char.origin];      // 门阀 1.5 / 外戚 1.8 / 科举 1.0 / 寒门 0.7 / 荫叙 1.2
  var austerityFactor = DYNASTY_AUSTERITY[sc.dynasty];  // 越薄俸越低
  var regionWealth = REGION_WEALTH[char.birthplace];     // 江南 1.5 / 朔方 0.5
  
  var baseSalary = dynastyBase.salary * rankMultiplier;
  var baseCash = baseSalary * 3 * originMultiplier * regionWealth / austerityFactor;
  
  // 名望初值
  var fame = (char.rank <= 3 ? 20 : 0)
           + (char.origin === '门阀' ? 30 : 0)
           + (char.character === '清廉' ? 25 : 0)
           - (char.character === '贪婪' ? 15 : 0);
  
  // 贤能初值（按履历估算）
  var virtueMerit = _estimateFromTenure(char.careerHistory)
                 || (char.rank === 1 ? 400 : (char.rank <= 3 ? 200 : 50));
  
  return {
    publicTreasury: _deriveFromPosition(char.position, dynastyBase),
    privateWealth: {
      cash: { money: baseCash, grain: baseCash * 0.3, cloth: 0 },
      land: _estimateLand(char, regionWealth),
      houses: _estimateHouses(char.rank),
      shops: [],
      treasures: _estimateTreasures(char.rank, char.origin)
    },
    fame,
    virtueMerit,
    virtueTier: _calcTier(virtueMerit),
    health: 90 - char.age * 0.5 + random(-5, 5),
    stress: 30 + (char.position ? 20 : 0) + random(-10, 10)
  };
}
```

**触发场景**（都需自动补全）：

2. **剧本预设人物在某回合进场**：如某角色到成年才出现
3. **玩家诏令新增**：如"擢拔某乡野贤才"

5. **继承/婚姻产生新对象**：家谱自动扩展

```js
function _ensureCharComplete(char, sc) {
  var missing = _detectMissingFields(char);      // 检测哪些字段缺失
  if (missing.length === 0) return;               // 完整则跳过
  
  // 三级补齐（与编辑器同）
  var filled = {};
  
  // Step 1: 历史档案库
  if (HISTORICAL_CHAR_PROFILES[char.name]) {
    Object.assign(filled, HISTORICAL_CHAR_PROFILES[char.name]);
  }
  
  // Step 2: 朝代/出身/官职 公式推算（填数值类+基础身份类）
  Object.assign(filled, _formulaEstimate(char, sc));
  
  // Step 3: 若剧本配置 AI API key 且 missing 中有散文类字段（bio/appearance/description/personality/恩怨等）
  if (sc.aiConfig && sc.aiConfig.key && _hasProseFields(missing)) {
    var aiResult = await _aiGenerateCharFields(char, missing, sc);
    Object.assign(filled, aiResult);
  }
  
  // Step 4: 随机波动 ±15%
  _applyRandomVariance(filled);
  
  // Step 5: 填回 char
  missing.forEach(function(field) {
    if (filled[field] !== undefined && char[field] === undefined) {
      char[field] = filled[field];
    }
  });
  
  // Step 6: 家谱补齐（如果缺亲属）
  if (!char._familyTreeComplete) _generateFamilyTree(char, sc);
  
  char._autoCompletedTurn = GM.turn;  // 标记
}
```

```js
  if (!char.gender) est.gender = random() < 0.92 ? '男' : '女';  // 古代官场为男性为主
  if (!char.age) est.age = 25 + Math.floor(randomGauss() * 20);
  if (!char.courtesy) est.courtesy = _generateCourtesyName(char.name);
  if (!char.ethnicity) est.ethnicity = sc.dynasty + '主族';
  if (!char.faith) est.faith = DYNASTY_FAITH_DEFAULT[sc.dynasty];
  if (!char.learning) est.learning = _inferLearningByRank(char.rank);
  if (!char.stance) est.stance = _inferStanceByPersonality(char);
  
  // 十维
  if (char.loyalty === undefined) est.loyalty = 50 + (char.playerRelation === '科举门生' ? 25 : 0);
  if (char.ambition === undefined) est.ambition = 30 + (char.occupation === '宰相' ? 30 : 0);
  // ... 其他 8 维类似
  
  // 资源
  if (!char.resources) est.resources = _estimateResources(char, sc);
  
  // 特质
  if (!char.traitIds) est.traitIds = _pickTraitsByTenMetrics(char);
```

## 「字」(courtesy name) 系统

### 生成规则

中文表字与名有语义关联。生成策略：

1. **同义扩展**：诸葛亮 字 孔明（亮=明 同义）
2. **反义互补**：朱熹 字 元晦（熹=亮 ↔ 晦=暗）
3. **补足义**：韩愈 字 退之（愈=进 ↔ 退）
4. **品德导向**：曹操 字 孟德（操=操行，德 承载）
5. **典故化**：关羽 字 云长（长羽=远翔）

### 算法实现

```js
function _generateCourtesyName(givenName) {
  // 取名字最后一字作为 anchor
  var anchor = givenName.slice(-1);
  
  // 1. 查预置词库（高频字→典型字配对）
  if (COURTESY_TEMPLATES[anchor]) {
    return _randomPick(COURTESY_TEMPLATES[anchor]);
  }
  
  // 2. 按语义匹配（同义/反义）
  var relatedChars = _semanticRelated(anchor);
  if (relatedChars.length) {
    var prefix = _randomPick(['子', '孟', '仲', '叔', '季', '伯', '元', '景', '德']);
    return prefix + _randomPick(relatedChars);
  }
  
  // 3. Claude API fallback
  if (sc.aiConfig?.key) {
    return _aiGenerateCourtesy(givenName, sc);
  }
  
  // 4. 最后退化：排行字 + 常见品德字
  return _randomPick(['子','德','敬','明','仲']) + _randomPick(['之','明','哲','清','直']);
}
```

### 游戏内使用规则（「字」的社交学）

```js
// 对话系统中的称呼选择
function _chooseAddressForm(speaker, target, context) {
  var relation = _getRelation(speaker, target);
  var formality = context.formality;  // 'formal' | 'informal' | 'intimate'
  
  // 1. 皇帝对臣
  if (speaker.isEmperor) {
    if (formality === 'formal') return target.name;      // 朝堂上称名
    if (formality === 'intimate') return target.courtesy; // 密谈称字（眷顾）
    return target.name;
  }
  
  // 2. 臣对皇帝
  if (target.isEmperor) return '陛下' || '圣上';
  
  // 3. 上下级
  if (relation.hierarchy === 'superior') return target.name;      // 称名（下对上）
  if (relation.hierarchy === 'subordinate') return target.officialTitle;  // 称官职（上对下）
  
  // 4. 平级
  if (relation.intimacy >= 0.7) return target.courtesy;    // 亲近 → 字
  if (relation.intimacy >= 0.3) return target.name;         // 平常 → 名
  if (relation.intimacy < 0) return target.surname + target.officialTitle;  // 敌对 → 姓+官职
  
  // 5. 家族内
  if (_sameFamily(speaker, target)) {
    if (_isElder(speaker, target)) return target.courtesy;  // 长辈对晚辈多称字
    else return _kinshipTerm(speaker, target);              // 晚辈用亲属称谓
  }
  
  // 6. 夫妻/挚友（极亲近）
  if (relation.intimacy >= 0.9) return target.courtesy;
  
  return target.name;
}
```

## 资源影响 NPC 行为决策（AI 联动详解）

### 六资源 → 行为权重模型

每个 NPC 在每回合 AI 推演时，其可能行为的概率都要受资源影响。系统将 6 资源转换为**行为倾向权重**注入 AI prompt。

```js
function _computeBehaviorWeights(char) {
  var res = char.resources;
  var priv = res.privateWealth.summary.totalValue;
  
  return {
    // 受贿概率：贫+债+野心 → 高
    briberySusceptibility: clamp(
      0.1
      + (priv.money < 0 ? 0.4 : 0)                      // 欠债大幅推高
      + (priv.money < dynastyBase.middleClass ? 0.2 : 0)
      + (char.ambition / 100) * 0.3,
      0, 1
    ),
    
    // 挪用公库概率
    embezzleTendency: clamp(
      0.05
      + (priv.money < 0 ? 0.3 : 0)
      + char.corruption * 0.5
      + (res.publicTreasury.quota.money > 100000 ? 0.1 : 0),  // 有的偷
      0, 1
    ),
    
    // 政治影响力（廷议话语权权重）
    politicalClout: clamp(
      0.2
      + (res.fame / 100) * 0.3
      + (res.virtueMerit / 1000) * 0.3
      + (priv.money / dynastyBase.very_rich) * 0.2,
      0, 2
    ),
    
    // 豪奢消费倾向（招惹御史）
    luxuryConsumption: clamp(
      0.2
      + (priv.money / dynastyBase.very_rich) * 0.4
      + (char.rank === 1 ? 0.2 : 0)
      + dynastyExtravagance * 0.3,
      0, 1
    ),
    
    // 党派资助能力
    partyFundingCapacity: clamp(
      (priv.money / dynastyBase.rich) * 0.5
      + (res.publicTreasury.quota.money / 1000000) * 0.3,
      0, 1
    ),
    
    // 反腐敏感度（越权贵越迟钝）
    antiCorruptionSensitivity: clamp(
      0.5
      - (res.fame < 0 ? 0.3 : 0)                        // 名声已坏，已无所谓
      - (char.clanInfluence / 100) * 0.3,
      0, 1
    ),
    
    // 离职风险（压力高 + 健康差 → 高）
    resignRisk: clamp(
      (res.stress / 100) * 0.4
      + ((100 - res.health) / 100) * 0.3
      + (res.fame < -50 ? 0.2 : 0),
      0, 1
    ),
    
    // 谋反风险（私产差+贤能高+名望-+健康好+兵权）
    rebelRisk: clamp(
      0.0
      + (priv.money < 0 ? 0.1 : 0)
      + (res.virtueMerit > 500 && res.fame < -30 ? 0.3 : 0)
      + (char.hasMilitaryPower ? 0.3 : 0)
      - (res.fame > 50 ? 0.2 : 0),                      // 名声好不愿反
      0, 1
    ),
    
    // 推荐人才能力（贤能+名望高）
    recruitTalent: clamp(
      (res.virtueMerit / 700) * 0.5 + (res.fame / 100) * 0.3,
      0, 1
    )
  };
}
```

| 情境 | 触发条件 | AI 可能行为 |
|------|---------|------------|

| NPC 举报同僚 | antiCorruptionSensitivity > 0.7 + 发现同僚腐败 | 上疏揭发 |
| NPC 请辞 | resignRisk > 0.75 + 皇帝允准 | 离任 |
| NPC 奢侈招祸 | luxuryConsumption > 0.8 | 御史弹劾 |

| NPC 党争资助 | partyFundingCapacity > 0.6 + 派系需要 | 暗中资助 |
| NPC 谋反 | rebelRisk > 0.7 + 军权 + 契机 | 起事 |
| NPC 隐退 | stress > 85 + age > 60 + 自主性格 | 告老归田 |

### 私产负值行为

**欠债角色**特殊行为激活：
- 更易接受贿赂（薪水不够养家）
- 更易挪用公库（短期周转）
- 更易参与贪腐集团（集体分赃）
- 更不敢得罪权贵（怕被追债）
- 可能向商人借贷（增加 debts 数组）
- 债务持续超过阈值 → 触发"破产"事件，被抄家或卖地

### 贤能阶位行为

不同阶位的 NPC 在 AI 叙事中有不同的"人设口吻"：

| 阶位 | AI 叙事特征 |
|------|-----------|
| 初入仕途 | 谨慎、少言、多请示 |
| 颇有政绩 | 开始有主见、对上级政策提出温和建议 |
| 地方重臣 | 能独当一面、有自己的施政纲领 |
| 朝廷要员 | 参与重大决策辩论、与宰相讨价还价 |
| 国之栋梁 | 主导改革、与皇帝共商国是 |
| 千古名臣 | 已入史书、言行被记录为典故 |

### 名望行为

名望为负的 NPC：
- AI 倾向把他们塑造为反面（"奸臣"）
- 更容易被御史弹劾
- 廷议时清流派反对
- 但权臣模式下仍可保有实权（李林甫/严嵩模式）

名望为正的 NPC：
- AI 倾向美化其言行
- 被攻讦时皇帝多袒护
- 廷议时朋友派力挺
- 但可能因刚直触怒皇帝（海瑞模式）

### 11.2 角色 integrity × 环境腐败（双向腐化）

角色清廉度随环境腐败波动：

```javascript
function applyEnvironmentalCorruptionToChar(char) {
  var deptCorruption = GM.corruption.subDepts[char.department].true;
  
  if (deptCorruption > 50) {
    // 清廉环境→ 清流稳定；腐败环境 → 加速堕落
    char.integrity -= (deptCorruption - 40) / 100 * 2;  // 每年最多 -2 integrity
    
    // integrity 极高者（> 80）反而可能坚持或辞官
    if (char.integrity > 80) {
      if (deptCorruption > 70) {
        // 辞官或被排挤
        scheduleEvent('integrityOfficialRetires', char);
      }
    }
  } else if (deptCorruption < 30) {
    // 清明环境：贪官反被约束
    if (char.integrity < 50) char.integrity += 0.5;
  }
}
```

```javascript
function selectPowerMinister() {
  var candidates = characters.filter(c => 
    c.influence > 75 &&
    c.loyaltyToEmperor < 60 &&
    (c.abilities.administration + c.abilities.intelligence) > 140 &&
    c.factionStrength > 50
  );
  
  // 若无符合，AI 晋升某高能力官员填补
  if (candidates.length === 0) {

}
```
